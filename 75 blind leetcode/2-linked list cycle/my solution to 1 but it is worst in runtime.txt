/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
// time o(n)
// memory o(n)
    bool hasCycle(ListNode *head) {
       
       ListNode *ptr;
       ptr = head;
       unordered_set <ListNode*> hashtable;
      while(ptr)
      {

       hashtable.insert(ptr);
       ptr=ptr->next;
       auto it = hashtable.find(ptr);
       if(it!=hashtable.end())
       return true;

      }
      return false;


    }
};
