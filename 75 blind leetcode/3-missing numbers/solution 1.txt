class Solution {
public:
//time complexity o(n) 
//memory o(n)
    int missingNumber(vector<int>& nums) 
    {
        unordered_set <int> s(nums.begin(),nums.end());
        for(int i= 0; i<=nums.size();i++)  //no nums[size], <= not <
        {
           auto it  = s.find(i);
           if(it==s.end())
           return i;


        }

       return 0;
    }
};
