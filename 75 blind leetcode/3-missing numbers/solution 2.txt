class Solution {
public:
// time o(n)
// memory o(1)
    int missingNumber(vector<int>& nums) 
    {
        int sumE=0;
        int sumI=0;
        for(int i = 0; i<nums.size(); i++)  // < not <= because we acsess nums[i]
        {
           sumE+=nums[i];
           sumI+=i;



        }
        sumI+=nums.size();
       return sumI-sumE;
    }
};
